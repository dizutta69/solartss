function doPost(e) { try { // 1. Obtener parámetros estructurados desde el formulario var idVisita = e.parameter.id_visita || "SIN_ID"; var tipoInstalacion = e.parameter.tipo_instalacion || ""; var nombreCliente = e.parameter.cliente || "Anonimo"; var direccion = e.parameter.direccion || ""; var gps = e.parameter.gps || ""; var tipoCubierta = e.parameter.tipo_cubierta || ""; var inclinacion = e.parameter.inclinacion || ""; var tipoRed = e.parameter.tipo_red || ""; var breakerPrincipal = e.parameter.breaker_principal || ""; // ID Corregido y validado de tu carpeta raíz de Drive var idCarpetaMaestra = "1KeEkJYhey5HyDJKFLCcR1JdO0Q8i8LG3"; // 2. Abrir carpeta raíz y crear la subcarpeta única de este cliente var carpetaMaestra = DriveApp.getFolderById(idCarpetaMaestra); var nombreCarpeta = "Survey_" + nombreCliente + "_" + idVisita; var carpetaCliente = carpetaMaestra.createFolder(nombreCarpeta); // 3. Crear el nuevo archivo de cálculo exclusivo para este levantamiento var nuevoSS = SpreadsheetApp.create("Reporte_Survey_" + nombreCliente + "_" + idVisita); var archivoSS = DriveApp.getFileById(nuevoSS.getId()); carpetaCliente.addFile(archivoSS); DriveApp.getRootFolder().removeFile(archivoSS); // Limpia la raíz de tu Drive // 4. Configurar Pestaña 1: Datos Técnicos (Formato Reporte Vertical) var hojaDatos = nuevoSS.getSheets()[0]; hojaDatos.setName("Datos"); hojaDatos.getRange("A1:B1").merge().setValue("REPORTE TÉCNICO - SOLAR TSS").setBackground("#1e3a8a").setFontColor("#ffffff").setFontWeight("bold").setHorizontalAlignment("center"); var filasDatos = [ ["ID de Visita:", idVisita], ["Tipo de Instalación:", tipoInstalacion], ["Cliente:", nombreCliente], ["Dirección del Sitio:", direccion], ["Coordenadas GPS:", gps], ["Tipo de Cubierta:", tipoCubierta], ["Inclinación (°):", inclinacion], ["Tipo de Red Eléctrica:", tipoRed], ["Breaker Principal:", breakerPrincipal], ["Enlace a Carpeta de Evidencias:", carpetaCliente.getUrl()] ]; for (var i = 0; i < filasDatos.length; i++) { hojaDatos.appendRow(filasDatos[i]); } // Estilos estéticos de la hoja de datos hojaDatos.getRange("A2:A11").setFontWeight("bold").setBackground("#f1f5f9"); hojaDatos.setColumnWidth(1, 220); hojaDatos.setColumnWidth(2, 420); hojaDatos.getRange("A1:B11").setBorder(true, true, true, true, true, true); // 5. Configurar Pestaña 2: Fotos incrustadas dentro de las celdas var hojaFotos = nuevoSS.insertSheet("Fotos"); hojaFotos.setColumnWidth(1, 250); hojaFotos.setColumnWidth(2, 400); hojaFotos.getRange("A1").setValue("Tipo de Registro").setFontWeight("bold").setBackground("#1e3a8a").setFontColor("#ffffff").setHorizontalAlignment("center"); hojaFotos.getRange("B1").setValue("Visualización de Foto (In-Cell)").setFontWeight("bold").setBackground("#1e3a8a").setFontColor("#ffffff").setHorizontalAlignment("center"); var tiposFotos = ["recibo", "tablero", "techo"]; var nombresVisuales = ["Foto Recibo de Luz", "Foto Tablero Eléctrico", "Foto Área de Techo"]; var filaActual = 2; for (var j = 0; j < tiposFotos.length; j++) { var fotoBytes = e.parameter["foto_" + tiposFotos[j]]; hojaFotos.getRange(filaActual, 1).setValue(nombresVisuales[j]).setFontWeight("bold").setVerticalAlignment("middle").setHorizontalAlignment("center"); hojaFotos.setRowHeight(filaActual, 220); // Alto de celda óptimo para renderizar la imagen if (fotoBytes && fotoBytes.length > 0) { var bytes = Utilities.base64Decode(fotoBytes); var blob = Utilities.newBlob(bytes, "image/jpeg", tiposFotos[j] + ".jpg"); var archivoFoto = carpetaCliente.createFile(blob); // Uso de la fórmula nativa IMAGE para incrustar la foto directo en la celda B hojaFotos.getRange(filaActual, 2).setFormula('=IMAGE("' + archivoFoto.getUrl() + '")'); } else { hojaFotos.getRange(filaActual, 2).setValue("No adjuntada en visita").setFontColor("#94a3b8").setVerticalAlignment("middle").setHorizontalAlignment("center"); } filaActual++; } hojaFotos.getRange("A1:B4").setBorder(true, true, true, true, true, true); // Responder con éxito rotundo al frontend en la web return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doOptions(e) { return ContentService.createTextOutput("") .setMimeType(ContentService.MimeType.TEXT) .setHeaders({ "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }); }
